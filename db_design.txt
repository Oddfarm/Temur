table user:
    id int nn pk uniq
    email varchar(64) nn uniq
    password varchar(255) nn
    is_active boolean nn default=FALSE
    activation_key varchar(255) nn uniq

table habbos:
    id int nn ck uniq
    user_id int nn ck uniq fk (habbos.user_id -> user.id)
    name varchar(32) nn uniq
    gender enum('M','F')
